CREATE TABLE CLIENT (
ID INT IDENTITY(1,1) PRIMARY KEY,
NOM NVARCHAR(50) NOT NULL,
PRENOM NVARCHAR(50) NOT NULL,
DATE_NAISSANCE DATE NOT NULL,
ADRESSE NVARCHAR(150) NULL,
CODE_POSTAL NVARCHAR(15) NULL,
VILLE NVARCHAR(50) NULL
);

CREATE TABLE MARQUE (
ID INT IDENTITY(1,1) PRIMARY KEY,
NOM NVARCHAR(50) NOT NULL,
UNIQUE(NOM)
)

CREATE TABLE CATEGORIE (
ID INT IDENTITY(1,1) PRIMARY KEY,
LIBELLE NVARCHAR(100) NOT NULL,
PRIX_KM INT NOT NULL,
UNIQUE(LIBELLE)
)

CREATE TABLE VEHICULE (
ID INT IDENTITY(1,1) PRIMARY KEY,
IMMATRICULATION NVARCHAR(50) NOT NULL,
MODELE VARCHAR(100) NOT NULL,
COULEUR NVARCHAR(50) NOT NULL,
ID_MARQUE INT FOREIGN KEY REFERENCES MARQUE(ID),
ID_CATEGORIE INT FOREIGN KEY REFERENCES CATEGORIE(ID),
UNIQUE(IMMATRICULATION)
)

CREATE TABLE LOUE (
ID INT IDENTITY(1,1) PRIMARY KEY,
ID_CLIENT INT FOREIGN KEY REFERENCES CLIENT(ID),
ID_VEHICULE INT FOREIGN KEY REFERENCES VEHICULE(ID),
NB_KM INT NOT NULL,
DATE_DEBUT DATE NOT NULL,
DATE_FIN DATE NULL
)
